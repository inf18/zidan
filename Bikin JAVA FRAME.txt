        // TODO add your handling code here:
        idoto();

    btntambah.setEnabled(true);
    btnbatal.setEnabled(true);
        }else if(btntambah.getText().equals("Simpan")){
            
            btntambah.setText("Tambah");
             try{
            String sql="Insert into customers values(?,?,?,?,?)";
            PreparedStatement st=(PreparedStatement) conn.prepareStatement(sql);
            if(txtidcust.getText().isEmpty() || txtnama.getText().isEmpty() || txtalamat.getText().isEmpty() || txtnohp.getText().isEmpty()){
                  JOptionPane.showMessageDialog(this,"Kolom Tidak Boleh Kosong");
              }
           //valdas
            
                st.setString(1, txtidcust.getText());
                st.setString(2, txtnama.getText());
                st.setString(3, txtalamat.getText());
                st.setString(4, txtnohp.getText());
                st.setObject(5, cbjenis.getSelectedItem());
            int rs=st.executeUpdate();
            if(rs>0){
            JOptionPane.showMessageDialog(this,"Input Berhasil");
      	    tampilDataKeJTable();
            }
        }
        catch (SQLException sqle) {
           System.out.println("Input  Gagal = " + sqle.getMessage());
        }
        catch(Exception e){
           System.out.println("Gagal " +e.getMessage());
           
        }
  }
    }
